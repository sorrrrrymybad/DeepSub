# DeepSub 前端实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建 DeepSub React Web 界面，包含任务中心、新建任务（SMB 文件浏览器）、设置管理和历史记录四个页面，通过 WebSocket 实时接收任务进度更新。

**Architecture:** React 18 + React Router 6 单页应用，Vite 构建，Tailwind CSS 样式，所有 API 调用封装在 `src/api/` 层，WebSocket 连接通过 Context 全局共享。页面按路由拆分，组件按功能划分（不按技术层）。

**Tech Stack:** React 18, React Router 6, Vite, Tailwind CSS, @tanstack/react-query（数据获取与缓存）, React Hook Form（表单）, dayjs（时间格式化）

**前置条件：** 后端已在 `http://localhost:8000` 运行。

// frontend/src/context/WebSocketContext.tsx
import React, { createContext, useContext, useEffect, useRef, useCallback } from 'react'
import { useQueryClient } from '@tanstack/react-query'

type Handler = (data: Record<string, unknown>) => void

interface WSContextType {
  subscribe: (taskId: number, handler: Handler) => () => void
}

const WSContext = createContext<WSContextType>({ subscribe: () => () => {} })

export function WebSocketProvider({ children }: { children: React.ReactNode }) {
  const queryClient = useQueryClient()
  const wsRef = useRef<WebSocket | null>(null)
  const handlersRef = useRef<Map<number, Set<Handler>>>(new Map())

  const connect = useCallback(() => {
    const ws = new WebSocket(`ws://${location.host}/ws/tasks`)
    wsRef.current = ws

    ws.onmessage = (event) => {
      try {
        const data = JSON.parse(event.data)
        const { task_id, ...rest } = data
        // 更新 react-query 缓存
        queryClient.invalidateQueries({ queryKey: ['tasks'] })
        // 通知订阅者
        handlersRef.current.get(task_id)?.forEach(h => h(rest))
      } catch {}
    }

    ws.onclose = () => {
      // 断线重连，3 秒后
      setTimeout(connect, 3000)
    }
  }, [queryClient])

  useEffect(() => {
    connect()
    return () => { wsRef.current?.close() }
  }, [connect])

  const subscribe = useCallback((taskId: number, handler: Handler) => {
    if (!handlersRef.current.has(taskId)) {
      handlersRef.current.set(taskId, new Set())
    }
    handlersRef.current.get(taskId)!.add(handler)
    return () => { handlersRef.current.get(taskId)?.delete(handler) }
  }, [])

  return <WSContext.Provider value={{ subscribe }}>{children}</WSContext.Provider>
}

export const useWebSocket = () => useContext(WSContext)
```

// frontend/src/components/TaskCard.tsx
import { useState } from 'react'
import type { Task } from '../api/tasks'
import { tasksApi } from '../api/tasks'
import { useQueryClient } from '@tanstack/react-query'
import StatusBadge from './StatusBadge'
import TaskLogDrawer from './TaskLogDrawer'
import dayjs from 'dayjs'

export default function TaskCard({ task }: { task: Task }) {
  const qc = useQueryClient()
  const [showLog, setShowLog] = useState(false)

  const duration = task.started_at && task.finished_at
    ? dayjs(task.finished_at).diff(dayjs(task.started_at), 'second') + 's'
    : null

  const handleCancel = async () => {
    await tasksApi.cancel(task.id)
    qc.invalidateQueries({ queryKey: ['tasks'] })
  }

  const handleRetry = async () => {
    await tasksApi.retry(task.id)
    qc.invalidateQueries({ queryKey: ['tasks'] })
  }

  const filename = task.file_path.split('/').pop() ?? task.file_path

  return (
    <div className="bg-white rounded-lg border p-4 flex flex-col gap-2">
      <div className="flex items-start justify-between gap-2">
        <div className="flex-1 min-w-0">
          <p className="font-medium text-sm truncate" title={task.file_path}>{filename}</p>
          <p className="text-xs text-gray-400 truncate">{task.file_path}</p>
        </div>
        <div className="flex items-center gap-2 shrink-0">
          <StatusBadge status={task.status} />
          {duration && <span className="text-xs text-gray-400">{duration}</span>}
        </div>
      </div>

      {task.status === 'running' && (
        <div className="w-full bg-gray-100 rounded-full h-1.5">
          <div className="bg-blue-500 h-1.5 rounded-full transition-all"
               style={{ width: `${task.progress}%` }} />
        </div>
      )}

      {task.error_message && (
        <p className="text-xs text-red-500 bg-red-50 rounded px-2 py-1">{task.error_message}</p>
      )}

      <div className="flex gap-2 text-xs">
        <button onClick={() => setShowLog(true)}
                className="text-blue-600 hover:underline">日志</button>
        {task.status === 'running' && (
          <button onClick={handleCancel} className="text-gray-500 hover:underline">取消</button>
        )}
        {task.status === 'failed' && task.retry_count < 5 && (
          <button onClick={handleRetry} className="text-orange-500 hover:underline">重试</button>
        )}
      </div>

      {showLog && <TaskLogDrawer taskId={task.id} onClose={() => setShowLog(false)} />}
    </div>
  )
}

// frontend/src/pages/TasksPage.tsx
import { useState } from 'react'
import { useQuery, useQueryClient } from '@tanstack/react-query'
import { tasksApi } from '../api/tasks'
import TaskCard from '../components/TaskCard'

const STATUS_FILTERS = [
  { label: '全部', value: '' },
  { label: '等待中', value: 'pending' },
  { label: '处理中', value: 'running' },
  { label: '完成', value: 'done' },
  { label: '失败', value: 'failed' },
]

export default function TasksPage() {
  const [status, setStatus] = useState('')
  const [page, setPage] = useState(1)
  const qc = useQueryClient()

  const { data, isLoading } = useQuery({
    queryKey: ['tasks', status, page],
    queryFn: () => tasksApi.list({ status: status || undefined, page, page_size: 20 }),
    refetchInterval: 5000, // 兜底轮询
  })

  const handleBatchCancel = async () => {
    const running = data?.items.filter(t => t.status === 'running') ?? []
    await Promise.all(running.map(t => tasksApi.cancel(t.id)))
    qc.invalidateQueries({ queryKey: ['tasks'] })
  }

  return (
    <div className="max-w-4xl mx-auto">
      <div className="flex items-center justify-between mb-4">
        <h1 className="text-xl font-bold">任务中心</h1>
        <button onClick={handleBatchCancel}
                className="text-sm text-gray-500 hover:text-red-500">
          取消全部运行中
        </button>
      </div>

      <div className="flex gap-2 mb-4">
        {STATUS_FILTERS.map(f => (
          <button key={f.value}
                  onClick={() => { setStatus(f.value); setPage(1) }}
                  className={`px-3 py-1 rounded-full text-sm border transition-colors ${
                    status === f.value
                      ? 'bg-blue-600 text-white border-blue-600'
                      : 'border-gray-200 text-gray-600 hover:bg-gray-50'
                  }`}>
            {f.label}
          </button>
        ))}
      </div>

      {isLoading ? (
        <p className="text-sm text-gray-400">加载中...</p>
      ) : (
        <>
          <div className="flex flex-col gap-3">
            {data?.items.map(task => <TaskCard key={task.id} task={task} />)}
            {data?.items.length === 0 && (
              <p className="text-sm text-gray-400 text-center py-8">暂无任务</p>
            )}
          </div>
          {data && data.total > data.page_size && (
            <div className="flex justify-center gap-2 mt-4">
              <button disabled={page === 1} onClick={() => setPage(p => p - 1)}
                      className="px-3 py-1 border rounded text-sm disabled:opacity-40">上一页</button>
              <span className="text-sm text-gray-500 px-2 py-1">
                {page} / {Math.ceil(data.total / data.page_size)}
              </span>
              <button disabled={page * data.page_size >= data.total}
                      onClick={() => setPage(p => p + 1)}
                      className="px-3 py-1 border rounded text-sm disabled:opacity-40">下一页</button>
            </div>
          )}
        </>
      )}
    </div>
  )
}
```

    const newPath = `${path.endsWith('/') ? path : path + '/'}${entry.name}`
    setHistory(h => [...h, path])
    setPath(newPath)
  }

  const goBack = () => {
    const prev = history.at(-1) ?? '/'
    setHistory(h => h.slice(0, -1))
    setPath(prev)
  }

  const isVideo = (name: string) => VIDEO_EXTS.some(ext => name.toLowerCase().endsWith(ext))

  const getFullPath = (name: string) => `${path.endsWith('/') ? path : path + '/'}${name}`

  return (
    <div className="border rounded-lg overflow-hidden">
      <div className="flex items-center gap-2 px-3 py-2 bg-gray-50 border-b text-sm">
        <button onClick={goBack} disabled={history.length === 0}
                className="text-blue-600 disabled:opacity-40">← 返回</button>
        <span className="text-gray-500 truncate flex-1">{path}</span>
      </div>
      <div className="divide-y max-h-72 overflow-auto">
        {isLoading && <p className="text-sm text-gray-400 p-3">加载中...</p>}
        {data?.map(entry => {
          const fullPath = getFullPath(entry.name)
          const isSelected = selected.includes(fullPath)
          return (
            <div key={entry.name}
                 className="flex items-center gap-2 px-3 py-2 hover:bg-gray-50 text-sm">
              {entry.is_dir ? (
                <button onClick={() => navigate(entry)}
                        className="flex items-center gap-2 flex-1 text-left">
                  <span>📁</span>
                  <span>{entry.name}</span>
                </button>
              ) : isVideo(entry.name) ? (
                <label className="flex items-center gap-2 flex-1 cursor-pointer">
                  <input type="checkbox" checked={isSelected}
                         onChange={() => onToggle(fullPath)} />
                  <span>🎬</span>
                  <span className="flex-1">{entry.name}</span>
                  <span className="text-gray-400 text-xs">
                    {(entry.size / 1024 / 1024).toFixed(0)} MB
                  </span>
                </label>
              ) : (
                <span className="text-gray-300 flex items-center gap-2 flex-1">
                  <span>📄</span>
                  <span>{entry.name}</span>
                </span>
              )}
            </div>
          )
        })}
      </div>
    </div>
  )
}

                className="w-full bg-blue-600 text-white py-2 rounded-lg text-sm font-medium

          {submitting ? '提交中...' : `提交 ${selectedFiles.length} 个任务`}
        </button>
      </div>
    </div>
  )
}
