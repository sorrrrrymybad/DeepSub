# DeepSub 后端实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建 DeepSub 后端服务，包含 FastAPI REST API、Celery 任务队列、SMB 文件访问、字幕提取、STT 转写和翻译引擎抽象层。

**Architecture:** FastAPI 提供 HTTP/WebSocket 接口，Celery Worker 异步处理翻译任务，引擎层通过统一抽象接口隔离 STT 和翻译实现，SQLAlchemy + SQLite 持久化数据。视频文件先从 SMB 下载到本地 tmp 目录再处理，任务完成后立即清理临时文件。

**Tech Stack:** Python 3.11+, FastAPI, SQLAlchemy 2.x, Celery 5.x, Redis, smbprotocol, ffmpeg/ffprobe (系统命令), faster-whisper, cryptography (Fernet), pytest, httpx

```python
# backend/core/crypto.py
import os
import base64
from cryptography.fernet import Fernet
from core.config import settings

def _get_fernet() -> Fernet:
    key_bytes = settings.secret_key.encode()
    # Fernet 需要 32 字节 base64url 编码的密钥
    padded = key_bytes[:32].ljust(32, b"0")
    key = base64.urlsafe_b64encode(padded)
    return Fernet(key)

def encrypt(plaintext: str) -> str:
    return _get_fernet().encrypt(plaintext.encode()).decode()

def decrypt(token: str) -> str:
    return _get_fernet().decrypt(token.encode()).decode()

def mask_secret(value: str) -> str:
    if len(value) <= 4:
        return "****"
    return "****" + value[-4:]
```

```python
# backend/smb/client.py
from __future__ import annotations
import os
from typing import Optional
from smbprotocol.connection import Connection
from smbprotocol.session import Session
from smbprotocol.tree import TreeConnect
from smbprotocol.open import Open, CreateDisposition, FileAttributes, ShareAccess, CreateOptions, FilePipePrinterAccessMask
from smbprotocol.query_info import QueryInfoRequest, InfoType, FileInformationClass
import uuid


class SMBConnectionError(Exception):
    pass


class SMBClient:
    def __init__(self, host: str, port: int, share: str,
                 username: str, password: str, domain: Optional[str] = None):
        self.host = host
        self.port = port
        self.share = share
        self.username = username
        self.password = password
        self.domain = domain or ""
        self._conn = None
        self._session = None
        self._tree = None

    def connect(self):
        try:
            import smbclient
            smbclient.register_session(
                self.host,
                username=self.username,
                password=self.password,
                port=self.port,
            )
            self._smbclient = smbclient
        except Exception as e:
            raise SMBConnectionError(f"Cannot connect to {self.host}: {e}") from e

    def _smb_path(self, remote_path: str) -> str:
        path = remote_path.lstrip("/")
        return f"\\\\{self.host}\\{self.share}\\{path.replace('/', '\\')}"

    def list_directory(self, remote_path: str) -> list[dict]:
        import smbclient
        smb_path = self._smb_path(remote_path)
        entries = []
        for entry in smbclient.scandir(smb_path):
            entries.append({
                "name": entry.name,
                "is_dir": entry.is_dir(),
                "size": entry.stat().st_size if not entry.is_dir() else 0,
            })
        return entries

    def download_file(self, remote_path: str, local_path: str) -> None:
        import smbclient
        smb_path = self._smb_path(remote_path)
        os.makedirs(os.path.dirname(local_path), exist_ok=True)
        with smbclient.open_file(smb_path, mode="rb") as src:
            with open(local_path, "wb") as dst:
                while chunk := src.read(1024 * 1024):
                    dst.write(chunk)

    def upload_file(self, local_path: str, remote_path: str) -> None:
        import smbclient
        smb_path = self._smb_path(remote_path)
        with open(local_path, "rb") as src:
            with smbclient.open_file(smb_path, mode="wb") as dst:
                while chunk := src.read(1024 * 1024):
                    dst.write(chunk)

    def file_exists(self, remote_path: str) -> bool:
        import smbclient
        try:
            smbclient.stat(self._smb_path(remote_path))
            return True
        except Exception:
            return False

    @classmethod
    def from_server_model(cls, server) -> "SMBClient":
        from core.crypto import decrypt
        return cls(
            host=server.host,
            port=server.port,
            share=server.share,
            username=server.username,
            password=decrypt(server.password),
            domain=server.domain,
        )
```

```python
# backend/worker/subtitle_extractor.py
from __future__ import annotations
import json
import subprocess
from dataclasses import dataclass
from typing import Optional


@dataclass
class SubtitleTrack:
    index: int
    language: Optional[str]
    forced: bool


def probe_subtitle_tracks(video_path: str) -> list[SubtitleTrack]:
    cmd = [
        "ffprobe", "-v", "quiet", "-print_format", "json",
        "-show_streams", "-select_streams", "s", video_path
    ]
    result = subprocess.run(cmd, capture_output=True, text=True, timeout=60)
    if result.returncode != 0:
        return []
    data = json.loads(result.stdout)
    tracks = []
    for stream in data.get("streams", []):
        tags = stream.get("tags", {})
        disposition = stream.get("disposition", {})
        lang = tags.get("language") or None
        if lang in ("und", ""):
            lang = None
        forced = bool(disposition.get("forced", 0))
        tracks.append(SubtitleTrack(
            index=stream["index"],
            language=lang,
            forced=forced,
        ))
    return tracks


def select_best_track(tracks: list[SubtitleTrack], source_lang: str) -> Optional[SubtitleTrack]:
    if not tracks:
        return None
    if source_lang == "auto":
        # 无语言偏好：排除 forced，选第一条；若全是 forced，选第一条
        non_forced = [t for t in tracks if not t.forced]
        return (non_forced or tracks)[0]

    # 优先匹配语言，排除 forced
    matched = [t for t in tracks if t.language == source_lang]
    non_forced_matched = [t for t in matched if not t.forced]
    if non_forced_matched:
        return non_forced_matched[0]
    if matched:
        return matched[0]
    # 无匹配语言时，按 auto 规则兜底
    return select_best_track(tracks, "auto")


def extract_subtitle_track(video_path: str, track_index: int, output_srt: str) -> None:
    cmd = [
        "ffmpeg", "-y", "-i", video_path,
        "-map", f"0:{track_index}",
        "-c:s", "srt",
        output_srt
    ]
    result = subprocess.run(cmd, capture_output=True, text=True, timeout=300)
    if result.returncode != 0:
        raise RuntimeError(f"ffmpeg extract failed: {result.stderr}")
```

```python
# backend/engines/base.py
from abc import ABC, abstractmethod
from typing import Optional


class TranslateEngine(ABC):
    @abstractmethod
    def translate(self, text: str, source_lang: str, target_lang: str) -> str:
        """翻译单段文本，返回翻译结果"""

    def translate_batch(self, texts: list[str], source_lang: str, target_lang: str) -> list[str]:
        """批量翻译，默认逐条调用 translate"""
        return [self.translate(t, source_lang=source_lang, target_lang=target_lang) for t in texts]


class STTEngine(ABC):
    @abstractmethod
    def transcribe(self, audio_path: str, language: Optional[str] = None) -> list[dict]:
        """
        转写音频，返回 segments 列表：
        [{"start": 0.0, "end": 1.5, "text": "Hello"}]
        """
```

```python
# backend/worker/srt_writer.py
from __future__ import annotations
import pysrt
from datetime import timedelta


def segments_to_srt(segments: list[dict], output_path: str) -> None:
    """将 segments 列表写成 SRT 文件"""
    subs = pysrt.SubRipFile()
    for i, seg in enumerate(segments, start=1):
        item = pysrt.SubRipItem(
            index=i,
            start=_seconds_to_time(seg["start"]),
            end=_seconds_to_time(seg["end"]),
            text=seg["text"],
        )
        subs.append(item)
    subs.save(output_path, encoding="utf-8")


def _seconds_to_time(seconds: float) -> pysrt.SubRipTime:
    td = timedelta(seconds=seconds)
    total_ms = int(td.total_seconds() * 1000)
    h, remainder = divmod(total_ms, 3_600_000)
    m, remainder = divmod(remainder, 60_000)
    s, ms = divmod(remainder, 1000)
    return pysrt.SubRipTime(h, m, s, ms)
```

```python
def _update_task(db, task_id: int, **kwargs):
    db.query(Task).filter(Task.id == task_id).update(kwargs)

def _build_translate_engine(engine_name: str, db):
    from models.setting import Setting
    from core.crypto import decrypt
    from engines.translate.deeplx import DeepLXEngine
    from engines.translate.deepl import DeepLEngine
    from engines.translate.google import GoogleTranslateEngine
    from engines.translate.openai_translate import OpenAITranslateEngine

    def get_setting(key):
        s = db.query(Setting).filter(Setting.key == key).first()
        if not s:
            return ""
        try:
            return decrypt(s.value)
        except Exception:
            return s.value  # 非加密值（如 endpoint、model 名）直接返回

    if engine_name == "deeplx":
        return DeepLXEngine(endpoint=get_setting("translate.deeplx.endpoint"))
    elif engine_name == "deepl":
        return DeepLEngine(api_key=get_setting("translate.deepl.api_key"))
    elif engine_name == "google":
        return GoogleTranslateEngine(api_key=get_setting("translate.google.api_key"))
    elif engine_name in ("openai", "claude"):
        return OpenAITranslateEngine(
            api_key=get_setting(f"translate.{engine_name}.api_key"),
            model=get_setting(f"translate.{engine_name}.model") or "gpt-4o-mini",
            base_url=get_setting(f"translate.{engine_name}.base_url") or None,

    from core.crypto import decrypt

@celery_app.task(bind=True, max_retries=0)
def process_subtitle_task(self, task_id: int):
    db = SessionLocal()
    log_path = _get_log_path(task_id)
    tmp_dir = _get_tmp_dir(task_id)
    os.makedirs(tmp_dir, exist_ok=True)
    os.makedirs(os.path.dirname(log_path), exist_ok=True)

    file_handler = logging.FileHandler(log_path)
    task_logger = logging.getLogger(f"task.{task_id}")
    task_logger.addHandler(file_handler)
    task_logger.setLevel(logging.INFO)

    try:
        task = db.query(Task).filter(Task.id == task_id).first()
        if not task:
            return

        _update_task(db, task_id, status="running", started_at=datetime.now(timezone.utc),
                     celery_task_id=self.request.id, progress=5)
        task_logger.info(f"Started processing: {task.file_path}")

        # 获取 SMB 服务器信息
        from models.smb_server import SMBServer
        server = db.query(SMBServer).filter(SMBServer.id == task.smb_server_id).first()
        client = SMBClient.from_server_model(server)

        # 下载视频到本地
        local_video = os.path.join(tmp_dir, "video" + Path(task.file_path).suffix)
        task_logger.info("Downloading video from SMB...")
        client.download_file(task.file_path, local_video)
        _update_task(db, task_id, progress=20)

        # 检测内置字幕
        tracks = probe_subtitle_tracks(local_video)
        task_logger.info(f"Found {len(tracks)} subtitle track(s)")

        if tracks:
            best = select_best_track(tracks, task.source_lang)
            task_logger.info(f"Extracting track index={best.index}, lang={best.language}")
            extracted_srt = os.path.join(tmp_dir, "extracted.srt")
            extract_subtitle_track(local_video, best.index, extracted_srt)
            _update_task(db, task_id, progress=40)

            import pysrt
            subs = pysrt.open(extracted_srt)
            segments = [{"start": s.start.ordinal / 1000, "end": s.end.ordinal / 1000,
                         "text": s.text} for s in subs]
        else:
            # 无字幕，用 STT
            task_logger.info("No subtitle tracks found, running STT...")
            audio_path = os.path.join(tmp_dir, "audio.wav")
            import subprocess
            subprocess.run(["ffmpeg", "-y", "-i", local_video, "-vn", "-ar", "16000",
                            "-ac", "1", audio_path], check=True, capture_output=True, timeout=1800)
            _update_task(db, task_id, progress=40)

            stt = _build_stt_engine(task.stt_engine, db)
            lang = task.source_lang if task.source_lang != "auto" else None
            segments = stt.transcribe(audio_path, language=lang)

        task_logger.info(f"Got {len(segments)} segments, starting translation...")
        _update_task(db, task_id, progress=60)

        # 翻译
        translator = _build_translate_engine(task.translate_engine, db)
        texts = [seg["text"] for seg in segments]
        translated = translator.translate_batch(texts, source_lang=task.source_lang, target_lang=task.target_lang)
        for seg, t in zip(segments, translated):
            seg["text"] = t
        _update_task(db, task_id, progress=85)

        # 生成 SRT
        output_srt_name = Path(task.file_path).stem + f".{task.target_lang}.srt"
        output_srt_remote = str(Path(task.file_path).parent / output_srt_name)

        # 检查是否已存在
        if not task.overwrite and client.file_exists(output_srt_remote):
            task_logger.info(f"SRT already exists, skipping (overwrite=False): {output_srt_remote}")
        else:
            local_srt = os.path.join(tmp_dir, output_srt_name)
            segments_to_srt(segments, local_srt)
            task_logger.info(f"Uploading SRT to SMB: {output_srt_remote}")
            client.upload_file(local_srt, output_srt_remote)

        _update_task(db, task_id, status="done", progress=100,
                     finished_at=datetime.now(timezone.utc))
        task_logger.info("Task completed successfully.")

    except Exception as e:
        task_logger.exception(f"Task failed: {e}")
        _update_task(db, task_id, status="failed",
                     error_message=str(e)[:500],
                     finished_at=datetime.now(timezone.utc))
    finally:
        shutil.rmtree(tmp_dir, ignore_errors=True)
        file_handler.close()
        db.close()
```

```python
from __future__ import annotations
import logging
from celery_app import celery_app
from core.database import SessionLocal
from models.scheduled_job import ScheduledJob
from models.task import Task
from smb.client import SMBClient
from datetime import datetime, timezone

logger = logging.getLogger(__name__)

VIDEO_EXTENSIONS = {".mkv", ".mp4", ".avi", ".ts", ".mov"}


@celery_app.task
def run_scheduled_job(job_id: int):
    db = SessionLocal()
    try:
        job = db.query(ScheduledJob).filter(ScheduledJob.id == job_id, ScheduledJob.enabled == True).first()
        if not job:
            return

        from models.smb_server import SMBServer
        server = db.query(SMBServer).filter(SMBServer.id == job.smb_server_id).first()
        client = SMBClient.from_server_model(server)

        entries = client.list_directory(job.directory)
        video_files = [e for e in entries if not e["is_dir"]
                       and any(e["name"].lower().endswith(ext) for ext in VIDEO_EXTENSIONS)]

        created = 0
        for entry in video_files:
            file_path = f"{job.directory.rstrip('/')}/{entry['name']}"
            # 检查是否已有相同路径且非 failed 的任务
            existing = db.query(Task).filter(
                Task.smb_server_id == job.smb_server_id,
                Task.file_path == file_path,
                Task.target_lang == job.target_lang,
                Task.status.notin_(["failed", "cancelled"]),
            ).first()
            if existing:
                continue

            task = Task(
                smb_server_id=job.smb_server_id,
                file_path=file_path,
                source_lang=job.source_lang,
                target_lang=job.target_lang,
                stt_engine=job.stt_engine,
                translate_engine=job.translate_engine,
                overwrite=job.overwrite,
            )
            db.add(task)
            db.flush()

            from worker.subtitle_task import process_subtitle_task
            process_subtitle_task.delay(task.id)
            created += 1

        db.query(ScheduledJob).filter(ScheduledJob.id == job_id).update({
            "last_run_at": datetime.now(timezone.utc),
            "last_run_status": f"created {created} tasks",
        })
        db.commit()
    except Exception as e:
        logger.exception(f"Scheduled job {job_id} failed: {e}")
        db.query(ScheduledJob).filter(ScheduledJob.id == job_id).update({
            "last_run_at": datetime.now(timezone.utc),
            "last_run_status": f"error: {str(e)[:200]}",
        })
        db.commit()
    finally:
        db.close()
```

定时任务通过"每分钟轮询数据库，判断各 job 的 cron 是否到期"来驱动，无需第三方包。在 `tasks/scheduler.py` 末尾添加：

```python
# 安装依赖：pip install croniter
from croniter import croniter
from datetime import datetime, timezone, timedelta

@celery_app.task
def poll_scheduled_jobs():
    """每分钟由 Celery Beat 调用，检查哪些定时任务到期"""
    db = SessionLocal()
    try:
        jobs = db.query(ScheduledJob).filter(ScheduledJob.enabled == True).all()
        now = datetime.now(timezone.utc)
        for job in jobs:
            cron = croniter(job.cron_expr, job.last_run_at or (now - timedelta(minutes=2)))
            next_run = cron.get_next(datetime)
            if next_run <= now:
                run_scheduled_job.delay(job.id)
    finally:
        db.close()
```

```python
SENSITIVE_KEYS = {"api_key", "secret"}

def _is_sensitive(key: str) -> bool:
    return any(s in key for s in SENSITIVE_KEYS)

def _upsert(db, key: str, value: str):
    encrypted = encrypt(value) if _is_sensitive(key) else value
    setting = db.query(Setting).filter(Setting.key == key).first()
    if setting:
        setting.value = encrypted
    else:
        db.add(Setting(key=key, value=encrypted))

def _get_val(db, key: str) -> str | None:
    s = db.query(Setting).filter(Setting.key == key).first()
    if not s:
        return None
    from core.crypto import decrypt
    try:
        raw = decrypt(s.value)
    except Exception:
        raw = s.value
    return mask_secret(raw) if _is_sensitive(key) else raw
```

```python
# backend/api/ws.py
from __future__ import annotations
import asyncio
import json
from fastapi import APIRouter, WebSocket, WebSocketDisconnect

router = APIRouter(tags=["websocket"])

active_connections: set[WebSocket] = set()


@router.websocket("/ws/tasks")
async def websocket_tasks(websocket: WebSocket):
    await websocket.accept()
    active_connections.add(websocket)
    try:
        while True:
            # 保持连接存活，接收 ping
            await asyncio.wait_for(websocket.receive_text(), timeout=30)
    except (WebSocketDisconnect, asyncio.TimeoutError):
        pass
    finally:
        active_connections.discard(websocket)


async def broadcast_task_update(task_id: int, payload: dict):
    """由 Celery 任务通过 Redis pub/sub 触发，向所有 WS 客户端广播"""
    message = json.dumps({"task_id": task_id, **payload})
    dead = set()
    for ws in active_connections:
        try:
            await ws.send_text(message)
        except Exception:
            dead.add(ws)
    active_connections -= dead
```

```python
    await pubsub.subscribe("task_updates")
    async for message in pubsub.listen():
        if message["type"] == "message":
            data = json.loads(message["data"])
            await broadcast_task_update(data.pop("task_id"), data)
```
